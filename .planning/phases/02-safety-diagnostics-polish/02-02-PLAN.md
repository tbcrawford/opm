    ```go
    package cmd

    import (
        "fmt"
        "os"
        "text/tabwriter"

        "github.com/tbcrawford/opm/internal/paths"
        "github.com/tbcrawford/opm/internal/symlink"
        "github.com/spf13/cobra"
    )

    var doctorCmd = &cobra.Command{
        Use:          "doctor",
        Short:        "Check opm installation health",
        Args:         cobra.NoArgs,
        SilenceUsage: true,
        RunE:         runDoctor,
    }

    func init() {
        rootCmd.AddCommand(doctorCmd)
    }

    func runDoctor(cmd *cobra.Command, args []string) error {
        out := cmd.OutOrStdout()
        w := tabwriter.NewWriter(out, 0, 0, 2, ' ', 0)

        opencodeDir := paths.OpencodeConfigDir()
        s := newStore()

        warnings := 0
        failures := 0

        label := func(status, msg string) {
            fmt.Fprintf(w, "  [%s]\t%s\n", status, msg)
        }

        fmt.Fprintln(out, "opm health report:")

        // Check 1: Is opm managing ~/.config/opencode?
        managed, err := s.IsOpmManaged()
        if err != nil {
            label("FAIL", fmt.Sprintf("~/.config/opencode: %v", err))
            failures++
            _ = w.Flush()
            printSummary(out, warnings, failures)
            if failures > 0 {
                return fmt.Errorf("health check failed")
            }
            return nil
        }
        if !managed {
            label("FAIL", "~/.config/opencode is not an opm-managed symlink — run 'opm init'")
            failures++
            _ = w.Flush()
            printSummary(out, warnings, failures)
            return fmt.Errorf("health check failed")
        }

        // Check 2: Is the symlink dangling?
        st, err := symlink.Inspect(opencodeDir)
        if err != nil {
            label("FAIL", fmt.Sprintf("inspect ~/.config/opencode: %v", err))
            failures++
        } else if st.Dangling {
            label("FAIL", fmt.Sprintf("~/.config/opencode is a dangling symlink → %q (profile directory is missing)", st.Target))
            failures++
        } else {
            activeName, _ := s.ActiveProfile()
            label("OK", fmt.Sprintf("~/.config/opencode is an opm-managed symlink → %q", activeName))
        }

        // Check 3: Each profile directory.
        profiles, err := s.ListProfiles()
        if err != nil {
            label("FAIL", fmt.Sprintf("list profiles: %v", err))
            failures++
        } else {
            for _, p := range profiles {
                if p.Dangling {
                    label("FAIL", fmt.Sprintf("Profile %q directory is missing (dangling symlink)", p.Name))
                    failures++
                    continue
                }
                fi, statErr := os.Lstat(p.Path)
                if statErr != nil || !fi.IsDir() {
                    label("FAIL", fmt.Sprintf("Profile %q path is not a valid directory: %s", p.Name, p.Path))
                    failures++
                } else {
                    label("OK", fmt.Sprintf("Profile %q directory exists", p.Name))
                }
            }
        }

        // Check 4: Cross-check current file vs Readlink.
        current, curErr := s.GetCurrent()
        active, actErr := s.ActiveProfile()
        if curErr == nil && actErr == nil && current != "" && active != "" && current != active {
            label("WARN", fmt.Sprintf("current file says %q but active symlink points to %q", current, active))
            warnings++
        }

        _ = w.Flush()
        fmt.Fprintln(out)
        printSummary(out, warnings, failures)

        if failures > 0 {
            return fmt.Errorf("health check failed")
        }
        return nil
    }

    func printSummary(out interface{ Write([]byte) (int, error) }, warnings, failures int) {
        switch {
        case failures == 0 && warnings == 0:
            fmt.Fprintln(out, "Status: healthy")
        case failures == 0:
            fmt.Fprintf(out, "Status: %d warning(s)\n", warnings)
        default:
            fmt.Fprintf(out, "Status: %d failure(s), %d warning(s)\n", failures, warnings)
        }
    }
    ```

    **cmd/context_ls.go** — `opm context ls`:

    Behavior (per D-04, D-05):
    - Calls `s.ListProfiles()` — Profile.Active is derived from Readlink (authoritative)
    - Uses `text/tabwriter` for aligned output
    - Active profile: `* <name>`
    - Inactive profiles: `  <name>` (two spaces for alignment)
    - No path column, no size — clean and shell-prompt friendly
    - Empty profiles list: print nothing (or a hint like "No profiles found. Run 'opm context create <name>' to create one.")

    ```go
    package cmd

    import (
        "fmt"
        "os"
        "text/tabwriter"

        "github.com/<module>/internal/paths"
        "github.com/<module>/internal/store"
        "github.com/spf13/cobra"
    )

    var lsCmd = &cobra.Command{
        Use:               "ls",
        Short:             "List all profiles",
        Args:              cobra.NoArgs,
        PersistentPreRunE: managedGuard,
        RunE:              runContextLs,
        SilenceUsage:      true,
    }

    func init() {
        contextCmd.AddCommand(lsCmd)
    }

    func runContextLs(cmd *cobra.Command, args []string) error {
        s := newStore()
        profiles, err := s.ListProfiles()
        if err != nil {
            return err
        }
        if len(profiles) == 0 {
            fmt.Fprintln(cmd.OutOrStdout(), "No profiles found. Run 'opm context create <name>' to create one.")
            return nil
        }

        w := tabwriter.NewWriter(cmd.OutOrStdout(), 0, 0, 2, ' ', 0)
        for _, p := range profiles {
            marker := " "
            if p.Active {
                marker = "*"
            }
            fmt.Fprintf(w, "%s %s\n", marker, p.Name)
        }
        return w.Flush()
    }
    ```

    ```go
    var showCmd = &cobra.Command{
        Use:               "show",
        Short:             "Print the name of the currently active profile",
        Args:              cobra.NoArgs,
        PersistentPreRunE: managedGuard,
        RunE:              runContextShow,
        SilenceUsage:      true,
    }

    func init() {
        contextCmd.AddCommand(showCmd)
    }

    func runContextShow(cmd *cobra.Command, args []string) error {
        s := newStore()
        current, err := s.GetCurrent()
        if err != nil {
            return err
        }
        if current == "" {
            // Fallback: derive from Readlink if current file is absent
            current, err = s.ActiveProfile()
            if err != nil || current == "" {
                return fmt.Errorf("no active profile — run 'opm init' first")
            }
        }
        fmt.Fprintln(cmd.OutOrStdout(), current)
        return nil
    }
    ```

    **cmd/context_inspect.go** — `opm context inspect <name>`:

    Behavior (per D-06, D-07):
    - Shows: Name, Path, Active status, and first-level directory listing of the profile dir
    - No deep recursion — `os.ReadDir(profilePath)` only, one level
    - No model extraction from opencode.json in v1 (per D-07)
    - Directories shown with trailing `/`; files shown as-is

    Example output:
    ```
    Name:    work
    Path:    /Users/you/.config/opm/profiles/work
    Active:  true

    Contents:
      AGENTS.md
      agents/
      opencode.json
      plugins/
    ```

    Implementation:
    ```go
    var inspectCmd = &cobra.Command{
        Use:               "inspect <name>",
        Short:             "Show detailed information about a profile",
        Args:              cobra.ExactArgs(1),
        PersistentPreRunE: managedGuard,
        RunE:              runContextInspect,
        SilenceUsage:      true,
    }

    func init() {
        contextCmd.AddCommand(inspectCmd)
    }

    func runContextInspect(cmd *cobra.Command, args []string) error {
        name := args[0]
        s := newStore()

        profilePath, err := s.GetProfile(name)
        if err != nil {
            return err // "context "foo" does not exist"
        }

        active, _ := s.ActiveProfile()
        isActive := active == name

        out := cmd.OutOrStdout()
        fmt.Fprintf(out, "Name:    %s\n", name)
        fmt.Fprintf(out, "Path:    %s\n", profilePath)
        fmt.Fprintf(out, "Active:  %v\n", isActive)

        entries, err := os.ReadDir(profilePath)
        if err != nil {
            return fmt.Errorf("read profile contents: %w", err)
        }

        if len(entries) == 0 {
            fmt.Fprintln(out, "\nContents: (empty)")
            return nil
        }

        fmt.Fprintln(out, "\nContents:")
        for _, e := range entries {
            name := e.Name()
            if e.IsDir() {
                name += "/"
            }
            fmt.Fprintf(out, "  %s\n", name)
        }
        return nil
    }
    ```

    Register in `init()`: `contextCmd.AddCommand(inspectCmd)`.

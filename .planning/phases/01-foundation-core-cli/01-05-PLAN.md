    ```go
    package cmd

    import (
        "fmt"
        "sort"

        "github.com/<module>/internal/paths"
        "github.com/<module>/internal/symlink"
        "github.com/spf13/cobra"
    )

    var rmForce bool

    var rmCmd = &cobra.Command{
        Use:               "rm <name>",
        Short:             "Remove a profile",
        Args:              cobra.ExactArgs(1),
        PersistentPreRunE: managedGuard,
        RunE:              runContextRm,
        SilenceUsage:      true,
    }

    func init() {
        rmCmd.Flags().BoolVarP(&rmForce, "force", "f", false, "Force removal of the active profile (auto-switches first)")
        contextCmd.AddCommand(rmCmd)
    }

    func runContextRm(cmd *cobra.Command, args []string) error {
        name := args[0]
        s := newStore()

        // Validate the profile exists before any other work
        if _, err := s.GetProfile(name); err != nil {
            return err // "context "foo" does not exist"
        }

        // Determine if this is the active profile
        active, err := s.ActiveProfile()
        if err != nil {
            return fmt.Errorf("determine active profile: %w", err)
        }

        isActive := active == name

        if isActive && !rmForce {
            // Per SAFE-01 (guarded in Phase 1 base behavior): refuse without --force
            return fmt.Errorf("context %q is the active context.\n\nSwitch to another context first:\n  opm context use <name>\n\nOr force removal (auto-switches):\n  opm context rm --force %s", name, name)
        }

        if isActive && rmForce {
            // Per D-01: auto-switch before deleting
            switchTarget, err := selectAutoSwitchTarget(s, name)
            if err != nil {
                return err // includes D-03: "Cannot remove the only profile" error
            }

            // Switch symlink first, then delete
            targetDir := s.ProfileDir(switchTarget)
            if err := symlink.SetAtomic(targetDir, paths.OpencodeConfigDir()); err != nil {
                return fmt.Errorf("switch to %q: %w", switchTarget, err)
            }
            if err := s.SetCurrent(switchTarget); err != nil {
                return fmt.Errorf("update current: %w", err)
            }
            fmt.Fprintf(cmd.OutOrStdout(), "Switched to context %q.\n", switchTarget)
        }

        // Now delete the profile (no longer active)
        if err := s.DeleteProfile(name, true); err != nil {
            return err
        }
        fmt.Fprintf(cmd.OutOrStdout(), "Removed context %q.\n", name)
        return nil
    }

    // selectAutoSwitchTarget returns the profile to switch to when force-removing the active profile.
    // Per D-02: prefer "default" (if it exists and is not the deleted profile), then first alphabetical.
    // Per D-03: error if no other profile exists.
    func selectAutoSwitchTarget(s interface {
        ListProfiles() ([]store.Profile, error)
    }, deletingName string) (string, error) {
        profiles, err := s.ListProfiles()
        if err != nil {
            return "", err
        }

        // Collect candidates (exclude the profile being deleted)
        var candidates []string
        for _, p := range profiles {
            if p.Name != deletingName {
                candidates = append(candidates, p.Name)
            }
        }

        if len(candidates) == 0 {
            // D-03: no other profiles exist
            return "", fmt.Errorf("cannot remove the only profile. Create another profile first:\n  opm context create <name>")
        }

        // D-02: prefer "default" if available
        for _, c := range candidates {
            if c == "default" {
                return "default", nil
            }
        }

        // D-02: otherwise first alphabetically
        sort.Strings(candidates)
        return candidates[0], nil
    }
    ```

    Note: The `selectAutoSwitchTarget` interface parameter — use the concrete `*store.Store` type directly (simpler for v1, avoid premature abstraction). Import `store` package.
